from fractions import Fraction

class RationalSpinor:
    """Represents a discrete state on the lattice (A1). Must be a rational unit quaternion."""
    def __init__(self, w: Fraction, x: Fraction, y: Fraction, z: Fraction):
        self.w, self.x, self.y, self.z = w, x, y, z
        assert w**2 + x**2 + y**2 + z**2 == 1, "Spinor is not a unit quaternion!"
        assert w > 0, "Spinor w-component must be strictly positive."

class HyperbolicPumpKernel:
    def __init__(self, beta: Fraction, eta: Fraction):
        self.beta = beta  # Inverse Temperature
        self.eta = eta    # Torsional Resistance multiplier

    def lorentz_lift(self, q: RationalSpinor) -> tuple[Fraction, Fraction, Fraction, Fraction]:
        """Theorem T1: Closed Rational Lift. Exact stereographic projection to H^3."""
        v_x, v_y, v_z = q.x / (Fraction(1) + q.w), q.y / (Fraction(1) + q.w), q.z / (Fraction(1) + q.w)
        v_sq = v_x**2 + v_y**2 + v_z**2
        
        denom = Fraction(1) - v_sq # denom > 0 guaranteed because w > 0
        return (
            (Fraction(1) + v_sq) / denom,
            (Fraction(2) * v_x) / denom,
            (Fraction(2) * v_y) / denom,
            (Fraction(2) * v_z) / denom
        )

    def hyperbolic_quadrance(self, p: tuple, q: tuple) -> Fraction:
        """Theorem T2: Quadrance is a Rational Metric Kernel (Minkowski inner product)."""
        B_pq = (p*q) - (p*q) - (p*q) - (p*q)
        return Fraction(2) * self.beta * (B_pq - Fraction(1))

    def compute_logit(self, Q: Fraction, chirality: Fraction) -> Fraction:
        """Computes Executable Logit. chirality: 0 (forward), 1 (reverse)."""
        return Q / (Fraction(1) + self.eta * chirality)

    def sparsemax(self, logits: list[Fraction]) -> list[Fraction]:
        """Theorem T3: Rational Row-Stochastic Sparsity. Euclidean projection to simplex."""
        z_sorted = sorted(logits, reverse=True)
        k_val = 0
        sum_z = Fraction(0)
        
        for k in range(1, len(z_sorted) + 1):
            sum_z_temp = sum_z + z_sorted[k-1]
            if Fraction(1) + Fraction(k) * z_sorted[k-1] > sum_z_temp:
                k_val = k
                sum_z = sum_z_temp
            else: break
                
        tau = (sum_z - Fraction(1)) / Fraction(k_val)
        return [max(Fraction(0), z - tau) for z in logits]

    def apply_rational_valve(self, sparse_probs: list[Fraction], chiralities: list[Fraction]) -> list[Fraction]:
        """Theorem T5b: Executable Broken Balance. Routes rejected reverse mass to self-loop."""
        modified_probs = [Fraction(0)] * len(sparse_probs)
        rejected_mass = Fraction(0)
        
        for i in range(1, len(sparse_probs)):
            p, chi = sparse_probs[i], chiralities[i]
            sigma = max(Fraction(0), Fraction(1) - (self.eta * chi))
            
            forwarded_mass = p * sigma
            modified_probs[i] = forwarded_mass
            rejected_mass += (p - forwarded_mass)
            
        modified_probs = sparse_probs + rejected_mass # Index 0 is self-loop
        return modified_probs
